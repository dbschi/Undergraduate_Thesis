\section*{Preliminaries}
\subsection*{Number Theory Results}
\begin{theorem}[M\"obius Inversion]
	The M\"obius function $\mu$ is defined for $n\in\naturals$,
	\[
		\mu(n) = \begin{cases}
			1, \textrm{ if } n=1\\
			(-1)^k, \textrm{ if }n=p_1p_2...p_k\textrm{ for distinct }p\textrm{'s}\\
			0, \textrm {otherwise}
		\end{cases}
	\]
\end{theorem}
